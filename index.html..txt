/* ✅ FIX #8: تايمر يعتمد على startTime بدل العد التنازلي البحت */
var saiRunning=false;
var saiStartTime=null;
var saiPausedRemaining=25*60*1000; /* بالـ milliseconds */
var saiInterval=null;
var SAI_DURATION=25*60*1000;

function toggleTimer(){
 if(saiRunning){
  /* إيقاف مؤقت */
  clearInterval(saiInterval);
  saiPausedRemaining=saiStartTime?(saiPausedRemaining-(Date.now()-saiStartTime)):saiPausedRemaining;
  if(saiPausedRemaining<0)saiPausedRemaining=0;
  saiStartTime=null;
  saiRunning=false;
  document.getElementById('timer-btn').textContent='استئناف السعي';
 } else {
  /* بدء أو استئناف */
  if(saiPausedRemaining<=0){saiPausedRemaining=SAI_DURATION;}
  saiStartTime=Date.now();
  saiRunning=true;
  document.getElementById('timer-btn').textContent='إيقاف';
  saiInterval=setInterval(function(){
   var elapsed=Date.now()-saiStartTime;
   var remaining=saiPausedRemaining-elapsed;
   if(remaining<=0){
    clearInterval(saiInterval);
    saiRunning=false;saiStartTime=null;saiPausedRemaining=SAI_DURATION;
    updateTimerUIFromMs(0);
    document.getElementById('timer-btn').textContent='بدء السعي';
    var cur=gp('daily.'+td()+'.saiCount')||0;sp('daily.'+td()+'.saiCount',cur+1);
    var sel=document.getElementById('timer-task-sel');
    if(sel.value){var taskName=sel.options[sel.selectedIndex].text;toast('انتهت خلوة السعي لـ: '+taskName+' 🌿');}
    else{toast('انتهت خلوة السعي! 🌿');}
    renderSaiCount();
    /* صوت بسيط للتنبيه لو المتصفح بيدعم */
    try{var ctx=new(window.AudioContext||window.webkitAudioContext)();var o=ctx.createOscillator();var g=ctx.createGain();o.connect(g);g.connect(ctx.destination);o.type='sine';o.frequency.value=880;g.gain.setValueAtTime(0.3,ctx.currentTime);g.gain.exponentialRampToValueAtTime(0.001,ctx.currentTime+1);o.start();o.stop(ctx.currentTime+1);}catch(ex){}
   } else {
    updateTimerUIFromMs(remaining);
   }
  },250); /* تحديث كل 250ms بدل 1000ms للدقة */
 }
}
function updateTimerUIFromMs(remainingMs){
 var totalSec=Math.ceil(remainingMs/1000);
 var m=Math.floor(totalSec/60),s=totalSec%60;
 document.getElementById('timer-text').textContent=String(m).padStart(2,'0')+':'+String(s).padStart(2,'0');
 var pct=1-(remainingMs/SAI_DURATION);
 var ring=document.getElementById('timer-ring');
 ring.style.strokeDashoffset=Math.max(0,176-(176*pct));
 ring.style.stroke=pct>0.5?'var(--ok)':'var(--no)';
}

function calcStreak(){
 dbLoad();
 if(!_db.streak)_db.streak={};
 var s=_db.streak;
 var today=td();
 var cur=s.current||0;
 var last=s.lastDate||'';
 var best=s.best||0;
 var yest=new Date();yest.setDate(yest.getDate()-1);
 var yestStr=fds(yest);

 /* نشاط = مهمة منجزة واحدة على الأقل OR ورد الليلة */
 var todayActive=false;
 var todayTasks=gp('tasks.'+today)||[];
 if(todayTasks.some(function(t){return t.done;}))todayActive=true;
 if(_db.drHabits&&_db.drHabits[today])todayActive=true;

 if(todayActive){
  if(last===today){/* نفس اليوم — لا تغيير */}
  else if(last===yestStr){cur++;}
  else{cur=1;}
  s.lastDate=today;s.current=cur;s.best=Math.max(best,cur);
  dbSv();
 } else if(last&&last!==today&&last!==yestStr){
  s.current=0;s.lastDate='';dbSv();cur=0;
 }
 return {current:cur,best:s.best||0};
}

var taskDragId=null;
function initTaskDrag(){
 var items=document.querySelectorAll('#ul-tasks .ti[draggable]');
 items.forEach(function(el){
  /* Mouse drag */
  el.addEventListener('dragstart',function(e){
   taskDragId=this.dataset.id;
   setTimeout(function(){el.style.opacity='.35';},0);
   e.dataTransfer.effectAllowed='move';
  });
  el.addEventListener('dragend',function(){
   this.style.opacity='';
   document.querySelectorAll('.ti.drag-over-task').forEach(function(x){x.classList.remove('drag-over-task');});
   taskDragId=null;
  });
  el.addEventListener('dragover',function(e){
   e.preventDefault();
   if(this.dataset.id===taskDragId)return;
   document.querySelectorAll('.ti.drag-over-task').forEach(function(x){x.classList.remove('drag-over-task');});
   this.classList.add('drag-over-task');
  });
  el.addEventListener('dragleave',function(){this.classList.remove('drag-over-task');});
  el.addEventListener('drop',function(e){
   e.preventDefault();this.classList.remove('drag-over-task');
   if(!taskDragId||this.dataset.id===taskDragId)return;
   var tasks=getTasks();
   var fromIdx=tasks.findIndex(function(t){return String(t.id)===taskDragId;});
   var toIdx=tasks.findIndex(function(t){return String(t.id)===this.dataset.id;}.bind(this));
   if(fromIdx<0||toIdx<0)return;
   var moved=tasks.splice(fromIdx,1)[0];tasks.splice(toIdx,0,moved);
   setTasks(tasks);renderTasks();
  });

  /* ✅ FIX #6: Touch drag لـ iOS Safari */
  var touchStartY=0,touchStartId=null,touchClone=null;
  el.addEventListener('touchstart',function(e){
   touchStartId=this.dataset.id;
   touchStartY=e.touches[0].clientY;
   this.style.opacity='.6';
  },{passive:true});
  el.addEventListener('touchmove',function(e){
   if(!touchStartId)return;
   e.preventDefault();
   var touch=e.touches[0];
   var overEl=document.elementFromPoint(touch.clientX,touch.clientY);
   var overTask=overEl?overEl.closest('.ti[draggable]'):null;
   document.querySelectorAll('.ti.drag-over-task').forEach(function(x){x.classList.remove('drag-over-task');});
   if(overTask&&overTask.dataset.id!==touchStartId)overTask.classList.add('drag-over-task');
  },{passive:false});
  el.addEventListener('touchend',function(e){
   this.style.opacity='';
   var touch=e.changedTouches[0];
   var overEl=document.elementFromPoint(touch.clientX,touch.clientY);
   var overTask=overEl?overEl.closest('.ti[draggable]'):null;
   document.querySelectorAll('.ti.drag-over-task').forEach(function(x){x.classList.remove('drag-over-task');});
   if(overTask&&touchStartId&&overTask.dataset.id!==touchStartId){
    var tasks=getTasks();
    var fromIdx=tasks.findIndex(function(t){return String(t.id)===touchStartId;});
    var toIdx=tasks.findIndex(function(t){return String(t.id)===overTask.dataset.id;});
    if(fromIdx>=0&&toIdx>=0){var moved=tasks.splice(fromIdx,1)[0];tasks.splice(toIdx,0,moved);setTasks(tasks);renderTasks();}
   }
   touchStartId=null;
  },{passive:true});
 });
}
